# Use Node.js 22
FROM node:22

WORKDIR /app

# Copy dependency files and install
COPY package*.json ./
RUN npm install

# Copy app source code
COPY . .

# Build the Vite app
RUN npm run build

# Expose port for preview
EXPOSE 4173

# Environment variables for preview server
ENV HOST=0.0.0.0
ENV PORT=4173

# Run Vite preview
CMD ["npm", "run", "preview"]
